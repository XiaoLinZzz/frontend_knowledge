# 快速排序

快速排序就是确定一个 pivot （数组中间的数字）

	1. 小于 pivot 的数字加到 left
	1. 大于 pivot 的数字加到 right

然后递归进行处理。



```js
function quick_sorting(arr) {
  
		// Base case: an array of length 0 or 1 is already sorted
    if (arr.length <= 1) {
        return arr;
    }

    let left = [];
    let right = [];
    let mid = Math.floor(arr.length / 2);
    let pivot = arr[mid];

    for (let i = 0; i < arr.length; i++) {
        if (i !== Math.floor(arr.length / 2)) {
            if (arr[i] <= arr[mid]) {
                left.push(arr[i]);
            } else {
                right.push(arr[i]);
            }
        }
    }

    return [...quick_sorting(left), pivot, ...quick_sorting(right)];
}

let arr = [1,3,2,5,9,1];
console.log(quick_sorting(arr));
```
